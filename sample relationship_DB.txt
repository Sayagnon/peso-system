Table Register {
  reg_id int(11) [pk]
  username varchar(255)
  password varchar(255)
}

Table users {
  user_id int(11) [primary key]
  username varchar(255)
  password varchar(255)
  reg_id int(11) 
}

Table peso_client {
  p_cli_id integer [primary key]
  name varchar
  address varchar
  contact varchar
  agency_applied varchar
  ornumber int
  educ_attainment varchar

}

Table tesda_client {
  t_cli_id integer [primary key]
  name varchar
 address varchar
  contact varchar
  preferred_skill varchar
  other_details varchar
  employment_id int
  clearance_id int
}
Table clearance {
  
  cle_id integer 
  peso_client_id integer
  valid_until varchar
 

}

Table employment {
  emp_id integer
  peso_client_id integer
  certificate varchar
}

Table other_program {
  op_id integer
  peso_client_id integer
}


Table trainingskill {
  tra_id integer
  tesda_client_id integer
}

Table for_assesment {
  fa_id integer
  tesda_client_id integer
  result int
}

Ref: Register.reg_id < users.user_id // one-to-one

Ref: users.user_id < peso_client.p_cli_id // one-to-many // peso 

Ref: users.user_id < tesda_client.t_cli_id   // one-to-many // tesda

Ref: peso_client.p_cli_id > clearance.peso_client_id
Ref: peso_client.p_cli_id > employment.peso_client_id
Ref: peso_client.p_cli_id > other_program.peso_client_id

Ref:trainingskill.tesda_client_id < tesda_client.t_cli_id
Ref:for_assesment.tesda_client_id < tesda_client.t_cli_id
Ref: "tesda_client"."employment_id" < "employment"."emp_id"
Ref: "tesda_client"."clearance_id" < "clearance"."cle_id
